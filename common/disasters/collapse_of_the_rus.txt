collapse_of_the_rus = {
	potential = {
		tag = KIE
		NOT = { has_country_flag = had_rus_collapse }
		is_free_or_tributary_trigger = yes
		normal_or_historical_nations = yes
		principality = 5
	}
	
	can_start = {
		current_age = age_of_discovery
		is_at_war = no 
		has_any_disaster = no 
		is_lesser_in_union = no 
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			is_lesser_in_union = yes
			NOT = { government = monarchy }
		}
	}
	
	progress = {
		modifier = {
			factor = 1
			NOT = { legitimacy = 90 }
		}
		modifier = {
			factor = 1
			has_heir = no
		}
		modifier = {
			factor = 0.5
			NOT = { stability = 1 }
		}
		modifier = {
			factor = 0.5
			NOT = { ADM = 2 }
		}		
		modifier = {
			factor = 0.5
			NOT = { DIP = 2 }
		}		
		modifier = {
			factor = 0.5
			NOT = { MIL = 2 }
		}	
		modifier = {
			factor = 1
			NOT = { manpower_percentage = 0.75 }
		}
		modifier = {
			factor = 0.5
			NOT = { prestige = 5 }
		}
		modifier = {
			factor = 2
			any_country = {
				limit = {
					is_subject_of = ROOT
				}
				liberty_desire = 25
			}
		}
	}
	
	can_end = {
		has_global_flag = LOTE_KIE_ruthenian_collapse
		NOT = { num_of_rebel_controlled_provinces = 1 }
		stability = 2
		manpower_percentage = 0.5
		OR = {
			KIE = {
				all_subject_country = {
					NOT = {
						liberty_desire = 50
					}
				}
			}
			num_of_subjects = 0
		}
	}
	
	modifier = {
		stability_cost_modifier = 0.25
		prestige = -1
	}	
	
	on_start = LOTE_KIE_events.1
	on_end = LOTE_KIE_events.2
}
