country_event = { #Mishandling of Crusade
	id = lote_crusades.10
	title = lote_crusades.10.t 
	desc = lote_crusades.10.d 
	picture = ACCUSATION_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = VEN
		any_country = {
			religion = catholic
			defensive_war_with = ROOT
		}
		has_country_modifier = lote_crusades_ven_headlining_crusade
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	option = { #We should have known better.
		name = lote_crusades.10.a 
		if = {
			limit = {
				NOT = {
					is_excommunicated = yes
				}
			}
			excommunicate = ROOT
		}
		hidden_effect = {
			every_known_country = {
				limit = {
					religion = catholic
				}
				add_opinion = {
					who = ROOT
					modifier = lote_crusades_misused_crusaders
				}
			}
		}
		tooltip = lote_crusades_misused_crusaders_tt
		add_prestige = -25
		remove_country_modifier = lote_crusades_ven_headlining_crusade
	}
}
